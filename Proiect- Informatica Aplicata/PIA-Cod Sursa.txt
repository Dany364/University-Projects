void setupWiFi() {
  Serial.print("Conectare la "); Serial.println(ssid);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500); Serial.print(".");
  }
  Serial.println("\nConectat la rețea!");
}

  Serial.println("2 - 6K+3 (puncte/steluțe)");
  Serial.println("3 - 3K (alți pini GPIO)");

  while (!Serial.available()) delay(10);
  mode = Serial.readStringUntil('\n').toInt();

  if (mode == 3) {
    segmentPins[0] = 16; // A

    segmentPins[2] = 5;  // C
    segmentPins[3] = 18; // D
    segmentPins[4] = 19; // E

  }
}

int readIDFromSerial() {
  Serial.println("Introduceți ID episod:");
  while (!Serial.available()) delay(10);
  return Serial.readStringUntil('\n').toInt();
}

void getEpisodeData(int id) {
  HTTPClient http;
  String url = "http://proiectia.bogdanflorea.ro/api/south-park-episodes/episode?id=" + String(id);
  http.begin(url);

  int code = http.GET();
  if (code > 0) {
    String payload = http.getString();
    StaticJsonDocument<1024> doc;
    DeserializationError err = deserializeJson(doc, payload);
    if (!err) {
      episodeName = doc["name"].as<String>();
      description = doc["description"].as<String>();
      seasonNumber = doc["season"].as<int>();
      episodeNumber = doc["episode"].as<int>();
    }
  }
  http.end();
}

void clearDisplay() {
  for (int i = 0; i < 7; i++) digitalWrite(segmentPins[i], LOW);
  for (int i = 0; i < 2; i++) digitalWrite(digitPins[i], LOW);
}

void displayChar(char c, int digit) {
  clearDisplay();
  if (c >= 'a' && c <= 'z') c -= 32;
  if (c >= 'A' && c <= 'Z') {
    const byte* segs = letters[c - 'A'];
    for (int i = 0; i < 7; i++) digitalWrite(segmentPins[i], !segs[i]);
    digitalWrite(digitPins[digit], HIGH);
  } else if (c >= '0' && c <= '9') {
    byte num = c - '0';
    const byte digits[10] = {0x7E,0x30,0x6D,0x79,0x33,0x5B,0x5F,0x70,0x7F,0x7B};
    for (int i = 0; i < 7; i++) digitalWrite(segmentPins[i], !((digits[num] >> (6-i)) & 0x01));
    digitalWrite(digitPins[digit], HIGH);
  }
  delay(5);
}

void scrollText(String text, char symbol = 0) {
  text = " " + text + " ";
  for (int i = 0; i < text.length() - 1; i++) {
    unsigned long start = millis();
    while (millis() - start < 330) {
      displayChar(text[i], 0);   // DIGIT1 = GPIO2
      displayChar(text[i+1], 1); // DIGIT2 = GPIO13
    }
    clearDisplay();
    delay(80);

    if (mode == 2 && symbol != 0) Serial.print(symbol);
  }
  if (mode == 2 && symbol != 0) Serial.println();
  delay(1000);
}
